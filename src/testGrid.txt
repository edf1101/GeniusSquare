#include <Arduino.h>
#include <Adafruit_GFX.h>
#include <Adafruit_ST7789.h>
#include <SPI.h>
#include <RotaryWrapper.h>

// --- TFT Display Configuration ---
#define TFT_MOSI 5
#define TFT_SCLK 4
#define TFT_CS   11
#define TFT_DC   12
#define TFT_RST  13

#define ROT_A 17
#define ROT_B 18

RotaryWrapper rot(ROT_A, ROT_B);


SPIClass * hspi = new SPIClass(FSPI);
Adafruit_ST7789 tft = Adafruit_ST7789(hspi, TFT_CS, TFT_DC, TFT_RST);

uint16_t cols[] = {tft.color565(255, 0, 0), tft.color565(0, 255, 0), tft.color565(0, 0, 255), tft.color565(255, 255, 0), tft.color565(255, 0, 255), tft.color565(0, 255, 255)};
int colIndex = 0;


// --- Matrix Configuration ---
#define NUM_ROWS 6
#define NUM_COLS 6

int rowPins[NUM_ROWS] = { 14, 47, 48, 38, 41, 42 };
int colPins[NUM_COLS] = { 7, 8, 10, 9, 1, 2 };

// Settings for the color mapping
const int THRESHOLD = 500;
const int ADC_MAX = 4095; // ESP32 11dB attenuation max value

// Variables to store the calculated size of each grid cell
int cellWidth;
int cellHeight;

void changeCol(long change)
{
    // Change the column index and wrap around if it goes out of bounds
    colIndex += change;
    colIndex = ((colIndex%NUM_COLS) + NUM_COLS) % NUM_COLS; // Wrap around using modulo
}

void setup() {
    Serial.begin(115200);

    rot.setCallbackFunc(changeCol);
    // 1. Initialize TFT Display
    hspi->begin(TFT_SCLK, -1, TFT_MOSI, TFT_CS);
    tft.init(240, 280);
    tft.setRotation(1); // Usually makes it 280x240 landscape
    tft.fillScreen(ST77XX_BLACK);

    // Calculate cell dimensions based on screen size
    cellWidth = tft.width() / NUM_COLS;
    cellHeight = tft.height() / NUM_ROWS;

    // 2. Initialize Matrix ADC / Pins
    analogSetAttenuation(ADC_11db);
    pinMode(16,INPUT);

    for (int i = 0; i < NUM_ROWS; i++) {
        pinMode(rowPins[i], OUTPUT);
        digitalWrite(rowPins[i], HIGH);
    }

    for (int i = 0; i < NUM_COLS; i++) {
        pinMode(colPins[i], ANALOG);

        // Forcefully disable internal pull-ups and pull-downs
        gpio_pulldown_dis((gpio_num_t)colPins[i]);
        gpio_pullup_dis((gpio_num_t)colPins[i]);
    }

    Serial.println("System Ready: Matrix to TFT Active");
}

int readStableAnalog(int pin) {
    // Throwaway read to clear the internal ADC MUX/Capacitor
    analogRead(pin);
    delayMicroseconds(10);
    // Actual read
    return analogRead(pin);
}

// Function to calculate the interpolated color based on sensor value
uint16_t getMappedColor(int value) {
    int r, g, b;

    if (digitalRead(16)== HIGH)
    {
        return tft.color565(0, 255, 0); // Green for high signal on pin 16 (keep existing behaviour)
    }

    // Clamp value to prevent overflow math issues
    if (value < 0) value = 0;
    if (value > ADC_MAX) value = ADC_MAX;

    if (value <= THRESHOLD) {
        // Lerp from Black (0,0,0) to Grey (128,128,128)
        float t = (float)value / THRESHOLD;
        r = (int)(128 * t);
        g = (int)(128 * t);
        b = (int)(128 * t);
    } else {
        // Lerp from Grey (128,128,128) to the currently selected colour in cols[]
        float t = (float)(value - THRESHOLD) / (ADC_MAX - THRESHOLD);

        // Safely get the current target color from cols[] (stored as 16-bit RGB565)
        int colsCount = sizeof(cols) / sizeof(cols[0]);
        int idx = colIndex % colsCount;
        if (idx < 0) idx += colsCount;
        uint16_t target565 = cols[idx];

        // Extract 5/6/5 components and convert them to 0-255 range
        int tr = (target565 >> 11) & 0x1F;
        int tg = (target565 >> 5) & 0x3F;
        int tb = target565 & 0x1F;
        tr = (tr * 255 + 15) / 31; // scale to 0-255 with rounding
        tg = (tg * 255 + 31) / 63;
        tb = (tb * 255 + 15) / 31;

        // Interpolate from grey (128) to target color
        r = (int)(128 + (tr - 128) * t);
        g = (int)(128 + (tg - 128) * t);
        b = (int)(128 + (tb - 128) * t);
    }

    // Convert standard RGB to 16-bit 565 format used by Adafruit_GFX
    return tft.color565(r, g, b);
}
void loop() {
    rot.poll();

    for (int r = 0; r < NUM_ROWS; r++) {
        digitalWrite(rowPins[r], LOW); // Turn row ON

        // FIX 1: Increased Wake-up Delay.
        // 250us is often too fast for optical components. Let's bump it to 800us.
        delayMicroseconds(800);

        // FIX 2: The Row-Level Dummy Read.
        // Even with the delay, the ESP32's internal ADC multiplexer can be weird
        // immediately after a major voltage swing on the board. This primes it.
        analogRead(colPins[0]);
        delayMicroseconds(50);

        for (int c = 0; c < NUM_COLS; c++) {
            int val = readStableAnalog(colPins[c]);

            // Calculate color and draw
            uint16_t cellColor = getMappedColor(val);
            tft.fillRect(c * cellWidth, r * cellHeight, cellWidth, cellHeight, cellColor);
        }

        digitalWrite(rowPins[r], HIGH); // Turn row OFF

        // Give the columns time to drain to zero before hitting the next row
        delayMicroseconds(100);
    }

    // Minimal delay to prevent overwhelming the SPI bus
    delay(10);
}
